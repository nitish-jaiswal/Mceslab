#include <LPC21xx.H>
void clock_wise(void);
void anti_clock_wise(void);
unsigned long int var1;
unsigned int i=0,j=0,k=0;
int main(void)
{
PINSEL0 = 0x00000000; 
IO0DIR = 0x0000F000; 
while(1)
{
for(j=0;j<50;j++) 
 clock_wise();
for(k=0;k<65000;k++);
for(j=0;j<50;j++) 
anti_clock_wise();
for(k=0;k<65000;k++); 
} // End of while(1)
} // End of main
void clock_wise(void)
{
var1 = 0x00001000; 
 for(i=0;i<3;i++) 
{
IO0PIN = var1;
for(k=0;k<=30000;k++); 
var1 = var1<<1; 
 }
}
void anti_clock_wise(void)
{
var1 = 0x00008000; 
 for(i=0;i<3;i++) 
 {
 IO0PIN = var1;
for(k=0;k<=30000;k++); 
var1 = var1>>1;
 }
}
